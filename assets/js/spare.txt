   function displayAmenitiesData(park){
        $.getJSON(`https://developer.nps.gov/api/v1/parks?q=${park.name}&fields=hiking&api_key=${npsKey}`, function(data){
            console.log('in Here'); //check function call
            console.log(park)
            
            let activities = data.data[0].activities;
            //creating a model to display park activites data
            for(let i = 0; i < activities.length;i++){
                let parkActivities = activities[i].name;
                let activitiesHere = $('#aboutActivities');
                
                let listInActivites = document.createElement('ul');
                activitiesHere.append(listInActivites);
                
                let newActivity = document.createElement('li');
                newActivity.classList.add('modal')
                newActivity.textContent = parkActivities;
                activitiesHere.append(newActivity);            
                
                revealModal();
            }

        });
        
  
        
    }
    parkList.click(displayAmenitiesData);
